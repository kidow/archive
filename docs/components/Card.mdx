### Card

```tsx title="components/Card/index.tsx"
import type { FC, ReactNode } from 'react'
import classnames from 'classnames'
import type { Argument } from 'classnames'

import CardItem from './Item'

export interface Props extends ReactProps {
  title?: ReactNode
  caption?: ReactNode
  footer?: ReactNode
  border?: boolean
  padding?: boolean
  className?: Argument
}
interface State {}

const Card: FC<Props> = ({
  children,
  title,
  caption,
  footer,
  border = true,
  padding = true,
  className
}) => {
  return (
    <section className={classnames('bg-white shadow-md', { border: border })}>
      {!!title && (
        <div className="flex items-center justify-between p-6">
          <div className="font-bold">{title}</div>
          {!!caption && <div className="text-xs text-gray-400">{caption}</div>}
        </div>
      )}
      {!!children && (
        <div
          className={classnames(
            'relative',
            footer ? 'border-y' : 'border-t',
            { 'py-10 px-8': padding },
            className
          )}
        >
          {children}
        </div>
      )}
      {!!footer && <div className="py-4 px-8">{footer}</div>}
    </section>
  )
}

export default Object.assign(Card, { Item: CardItem })
```

#### Props

| Name    | Type        | Default |
| ------- | ----------- | ------- |
| title   | `ReactNode` |         |
| caption | `ReactNode` |         |
| footer  | `ReactNode` |         |
| border  | boolean     | true    |
| padding | boolean     | true    |

### Card.Item

```tsx title="components/Card/Item/index.tsx"
import type { FC, ReactNode } from 'react'
import classnames from 'classnames'

export interface Props extends ReactProps {
  label: ReactNode
  required?: boolean
  fullWidth?: boolean
  plain?: boolean
}
interface State {}

const CardItem: FC<Props> = ({
  children,
  label,
  required,
  fullWidth,
  plain
}) => {
  return (
    <div className="flex gap-10">
      <div
        className={classnames('min-w-[9rem] text-gray-400', {
          "whitespace-nowrap after:text-red-500 after:content-['*']": required
        })}
      >
        <div className="inline-flex h-[42px] items-center gap-0.5">{label}</div>
      </div>
      <div className={classnames({ 'flex-1': fullWidth, 'mt-2.5': plain })}>
        {children}
      </div>
    </div>
  )
}

export default CardItem
```

#### Props

| Name             | Type        | Default |
| ---------------- | ----------- | ------- |
| label<Required/> | `ReactNode` |         |
| required         | boolean     | false   |
| fullWidth        | boolean     | false   |
| plain            | boolean     | false   |
